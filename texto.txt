        let isAdminLoggedIn = false;

        let approvedComments = JSON.parse(localStorage.getItem('approved_comments')) || [];
        let pendingComments = JSON.parse(localStorage.getItem('pending_comments')) || [];

        window.onload = function() {
            renderApprovedComments();
            renderPendingComments();
            initSliders();
        };

        // LÓGICA DE FUNCIONAMIENTO PARA EL SLIDER DESLIZABLE
        const sliderStates = {};

        function initSliders() {
            document.querySelectorAll('.slider-container').forEach(slider => {
                const id = slider.id;
                sliderStates[id] = { currentIdx: 0 };
                
                // Soporte para deslizar con el dedo (Touch Swipe) en celular
                let startX = 0;
                slider.addEventListener('touchstart', e => startX = e.touches[0].clientX, {passive: true});
                slider.addEventListener('touchend', e => {
                    let diffX = startX - e.changedTouches[0].clientX;
                    if (Math.abs(diffX) > 50) {
                        moveSlide(id, diffX > 0 ? 1 : -1);
                    }
                }, {passive: true});
            });
        }

        function moveSlide(sliderId, direction) {
            const slider = document.getElementById(sliderId);
            const wrapper = slider.querySelector('.slider-wrapper');
            const slides = slider.querySelectorAll('.slide');
            const dots = slider.querySelectorAll('.dot');
            
            let state = sliderStates[sliderId];
            state.currentIdx += direction;

            if (state.currentIdx >= slides.length) state.currentIdx = 0;
            if (state.currentIdx < 0) state.currentIdx = slides.length - 1;

            wrapper.style.transform = `translateX(-${state.currentIdx * 100}%)`;

            // Actualizar puntito activo
            dots.forEach((dot, idx) => {
                dot.classList.toggle('active', idx === state.currentIdx);
            });
        }

        function toggleMenu() {
            const menu = document.getElementById('side-menu');
            if (menu) menu.classList.toggle('active');
        }

        function toggleFiltersPanel() {
            const panel = document.getElementById('filters-panel');
            const btn = document.getElementById('filter-trigger');
            panel.classList.toggle('show');
            btn.classList.toggle('active');
        }

        function handleAdminSession() {
            const adminPanel = document.getElementById('backoffice-panel');
            const navBtn = document.getElementById('admin-nav-btn');

            if (!isAdminLoggedIn) {
                const password = prompt("Ingrese la clave de Administrador:");
                if (password === "1234") {
                    alert("¡Acceso concedido, Anthony!");
                    isAdminLoggedIn = true;
                    adminPanel.classList.add('show');
                    navBtn.innerHTML = "🔓 Cerrar Sesión";
                    renderPendingComments();
                } else if (password !== null) {
                    alert("❌ Contraseña incorrecta.");
                }
            } else {
                isAdminLoggedIn = false;
                adminPanel.classList.remove('show');
                navBtn.innerHTML = "🔐 Acceso Admin";
                alert("Sesión cerrada.");
            }
        }

            if (name === "" || text === "") {
                alert("Por favor rellena tu nombre y el comentario.");
                return;
            }

            const newComment = { id: Date.now(), name, text, stars };
            pendingComments.push(newComment);
            localStorage.setItem('pending_comments', JSON.stringify(pendingComments));

            alert("¡Gracias! Tu reseña se mostrará en cuanto el administrador la apruebe.");
            document.getElementById('client-name').value = "";
            document.getElementById('client-text').value = "";
            renderPendingComments();
        }

        function renderApprovedComments() {
            const listContainer = document.getElementById('approved-comments-list');
            if (approvedComments.length === 0) {
                listContainer.innerHTML = `<p style="font-size:13px; color:#a1a1aa; text-align:center; padding: 15px 0;">No hay opiniones públicas todavía.</p>`;
                return;
            }
            let html = "";
            approvedComments.forEach(c => {
                html += `
                    <div class="comment-box">
                        <div class="comment-header">
                            <span class="comment-name">${c.name}</span>
                            <span class="comment-stars">${c.stars}</span>
                        </div>
                        <p class="comment-text">${c.text}</p>
                    </div>`;
            });
            listContainer.innerHTML = html;
        }

        function renderPendingComments() {
            const pendingContainer = document.getElementById('pending-comments-list');
            if (!pendingContainer) return;
            if (pendingComments.length === 0) {
                pendingContainer.innerHTML = `<p style="font-size:12px; color:#a1a1aa; text-align:center;">No hay comentarios pendientes.</p>`;
                return;
            }
            let html = "";
            pendingComments.forEach(c => {
                html += `
                    <div class="pending-box">
                        <strong>${c.name}</strong> (${c.stars})<br>
                        <span>"${c.text}"</span>
                        <div class="pending-actions">
                            <button class="btn-approve" onclick="approveComment(${c.id})">Aprobar</button>
                            <button class="btn-reject" onclick="rejectComment(${c.id})">Eliminar</button>
                        </div>
                    </div>`;
            });
            pendingContainer.innerHTML = html;
        }

        function approveComment(id) {
            const item = pendingComments.find(c => c.id === id);
            approvedComments.push(item);
            pendingComments = pendingComments.filter(c => c.id !== id);
            localStorage.setItem('approved_comments', JSON.stringify(approvedComments));
            localStorage.setItem('pending_comments', JSON.stringify(pendingComments));
            renderApprovedComments();
            renderPendingComments();
        }

        function rejectComment(id) {
            pendingComments = pendingComments.filter(c => c.id !== id);
            localStorage.setItem('pending_comments', JSON.stringify(pendingComments));
            renderPendingComments();
        }

        document.getElementById('search-input').addEventListener('input', function(e) {
            const query = e.target.value.toLowerCase().trim();
            document.querySelectorAll('.car-card').forEach(card => {
                const title = card.querySelector('.car-title').innerText.toLowerCase();
                card.style.display = title.includes(query) ? "block" : "none";
            });
        });

        function applyFilters() {
            const brandSel = document.getElementById('filter-brand').value;
            const modelSel = document.getElementById('filter-model').value;
            document.querySelectorAll('.car-card').forEach(card => {
                const b = card.getAttribute('data-marca');
                const m = card.getAttribute('data-modelo');
                let match = true;
                if (brandSel !== 'all' && b !== brandSel) match = false;
                if (modelSel !== 'all' && m !== modelSel) match = false;
                card.style.display = match ? "block" : "none";
            });
        }
